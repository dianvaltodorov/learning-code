На всяка стъпка има два варианта

f(i) = max(ai,  ai + f(i - 1))

f[0] = -4
f[1] = max ( f[0] + 6, 6) = 6
f[2] = max (6 + -7, - 7) = -1
f[3] = max (-1 + 8, 8) = 8
f[4] = max (8 + -5, - 5) = 3
f[5] = max (3 + 9, 9) = 12
